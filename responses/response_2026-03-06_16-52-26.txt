```python
import requests
from bs4 import BeautifulSoup
import csv

class MSEScraper:
    def __init__(self, url):
        self.url = url

    def fetch_data(self):
        """
        Fetch data from the given URL using an HTTP GET request.
        """
        try:
            response = requests.get(self.url)
            response.raise_for_status()  # Raise an error for bad status codes
            return response.text
        except requests.RequestException as e:
            print(f"Error fetching data: {e}")
            return None

    def parse_data(self, html_content):
        """
        Parse the HTML content to extract relevant data.
        """
        if not html_content:
            return []

        soup = BeautifulSoup(html_content, 'html.parser')
        table = soup.find('table')  # Assuming the data is in a table
        if not table:
            return []

        rows = table.find_all('tr')
        data = []
        for row in rows:
            cols = row.find_all('td')
            if cols:
                cols = [ele.text.strip() for ele in cols]
                data.append(cols)
        return data

    def save_to_csv(self, data, filename):
        """
        Save the extracted data to a CSV file.
        """
        with open(filename, 'w', newline='', encoding='utf-8') as file:
            writer = csv.writer(file)
            writer.writerows(data)

    def run(self):
        """
        Main function to run the scraper.
        """
        html_content = self.fetch_data()
        if html_content:
            data = self.parse_data(html_content)
            self.save_to_csv(data, 'output.csv')
            print("Data has been saved to output.csv")

if __name__ == "__main__":
    scraper = MSEScraper("http://example.com")  # Replace with the actual URL
    scraper.run()
```
